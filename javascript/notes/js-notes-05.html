Loops

For loop 

  for (let i = 0; i < 5; i++) {
    console.log(i);
  }
For loop in Reverse 

  for (let i = 5; i < 0; i--) {
    console.log(i);
  }
for loop iterates an array 

  const pets = ['cat', 'dog', 'fish'];

  for (let i = 0; i < pets.length; i++) {
    console.log(pets[i]);
  }
Nested Loops 

  for (let i = 2; i < 5; i++) {
    for (let j = 4; j < 7; j++) {
      console.log(`${i} * ${j} = ${i * j}`);
    }
  }
The While Loop 

  let counter = 1;

  while (counter < 4) {
    console.log(counter);
    counter++;
  }
Do...While Statements 

  let i = 0;

  do {
    console.log(i);
    i++;
  } while (i < 5);
The break Keyword

break exit the loop. 

  for (let i = 0; i < 10; i++) {
    if (i > 2) {
      break;
    }
    console.log(i);
  }
