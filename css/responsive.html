Responsive Web Design

MDN W3Schools

Viewport Meta Tag

<meta name="viewport" content="width=device-width, initial-scale=1">

Root Element Font Size

html {
  font-size: 16px;    /* default */
}

Responsive Units

width: 50%;
    
padding: 0.6em;
    
font-size: 1.4rem;

Min and Max Dimensions

min-width: 300px;
max-width: 600px;

min-height: 150px;
max-height: 300px;

Scaling Images and Videos

.container {
  width: 50%;
  height: 200px;
  overflow: hidden;
}

.container img {
  max-width: 100%;
  height: auto;
  display: block;
}

Scaling Background Images

body {
  backgpound-image: url('#');
  background-repeat: no-repeat;
  background-position: center;
  background-size: cover;
}

Media Queries

@media only screen and (max-width: 480px) {
  /* ruleset for <= 480px */
}
    
@media only screen and (min-width: 320px) {
  /* ruleset for >= 320px */
}
    
@media only screen and (min-width: 320px) and (max-width: 480px) {
  /* ruleset for 320px - 480px */
}
    
@media only screen and (min-width: 480px), (orientation: landscape) {
  /* ruleset for >= 480px or landscape orientated */
}

Resolution

dpi -- dots per inch
dpc -- dots per sentimeter
@media only screen and (min-resolution: 300dpi) {
  /* ruleset for high resolution screens */
}
    
@media only screen and (max-width: 480px) and (max-resolution: 150dpi) {
  /* ruleset for smalls low resolution screens */
}
